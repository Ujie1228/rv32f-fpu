// =====================================================================
// FPU CVT (类型转换) 模块测试向量表 (RISC-V 32-bit F 扩展)
// =====================================================================
// 格式说明:
// OP_TYPE    RM   DATA1     DATA2     DATA3     EXP_RESULT EXP_FLAGS
// (字符串)   (Hex)(Hex)     (Hex)     (Hex)     (Hex)      (Hex)
// ---------------------------------------------------------------------
// 异常标志位(Flags)对应 Hex 值:
// NV (无效)   = 10 (例如：转整数越界、NaN转换)
// NX (不精确) = 01 (例如：大整数转浮点丢失精度、浮点带小数转整数被截断)
// ---------------------------------------------------------------------

// ==========================================
// 1. FCVT.S.W (有符号整数转单精度浮点)
// DATA1 放入的是 32位整数 
// ==========================================
// 0 -> +0.0
FCVT_I2F_S    0    00000000  00000000  00000000  00000000   00
// 1 -> 1.0 (3F800000)
FCVT_I2F_S    0    00000001  00000000  00000000  3F800000   00
// -1 -> -1.0 (BF800000)
FCVT_I2F_S    0    FFFFFFFF  00000000  00000000  BF800000   00
// 大整数精度丢失: 0x7FFFFFFF (2^31-1) -> RNE舍入到 2^31 (4F000000), 触发 NX (01)
FCVT_I2F_S    0    7FFFFFFF  00000000  00000000  4F000000   01

// ==========================================
// 2. FCVT.S.WU (无符号整数转单精度浮点)
// ==========================================
// 0x80000000 (2^31) -> 2.1474836e9 (4F000000), 正确无精度丢失
FCVT_I2F_U    0    80000000  00000000  00000000  4F000000   00
// 大整数精度丢失: 0xFFFFFFFF (2^32-1) -> RNE舍入到 2^32 (4F800000), 触发 NX (01)
FCVT_I2F_U    0    FFFFFFFF  00000000  00000000  4F800000   01

// ==========================================
// 3. FCVT.W.S (单精度浮点转有符号整数)
// ==========================================
// +0.0 -> 0
FCVT_F2I_S    0    00000000  00000000  00000000  00000000   00
// 1.0 -> 1
FCVT_F2I_S    0    3F800000  00000000  00000000  00000001   00
// -1.0 -> -1
FCVT_F2I_S    0    BF800000  00000000  00000000  FFFFFFFF   00
// 小数舍入(RNE): 1.5 (3FC00000) -> 偶数 2, 触发 NX (01)
FCVT_F2I_S    0    3FC00000  00000000  00000000  00000002   01
// 小数舍入(RTZ): 1.5 (3FC00000) -> 趋零 1, 触发 NX (01)
FCVT_F2I_S    1    3FC00000  00000000  00000000  00000001   01
// 越界异常(正无穷): +Inf -> 饱和到 0x7FFFFFFF, 触发 NV (10)
FCVT_F2I_S    0    7F800000  00000000  00000000  7FFFFFFF   10
// 越界异常(负无穷): -Inf -> 饱和到 0x80000000, 触发 NV (10)
FCVT_F2I_S    0    FF800000  00000000  00000000  80000000   10
// 特殊数据(NaN): qNaN -> 饱和到 0x7FFFFFFF, 触发 NV (10)
FCVT_F2I_S    0    7FC00000  00000000  00000000  7FFFFFFF   10

// ==========================================
// 4. FCVT.WU.S (单精度浮点转无符号整数)
// ==========================================
// 2.0 -> 2
FCVT_F2I_U    0    40000000  00000000  00000000  00000002   00
// 越界异常(负数转无符号): -1.0 -> 负数转无符号在RISC-V中饱和到 0, 触发 NV (10)
FCVT_F2I_U    0    BF800000  00000000  00000000  00000000   10
// 越界异常(正无穷): +Inf -> 饱和到 0xFFFFFFFF, 触发 NV (10)
FCVT_F2I_U    0    7F800000  00000000  00000000  FFFFFFFF   10
// 越界异常(负无穷): -Inf -> 饱和到 0, 触发 NV (10)
FCVT_F2I_U    0    FF800000  00000000  00000000  00000000   10
// 特殊数据(NaN): qNaN -> 饱和到 0xFFFFFFFF, 触发 NV (10)
FCVT_F2I_U    0    7FC00000  00000000  00000000  FFFFFFFF   10
