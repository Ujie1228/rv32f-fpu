// =====================================================================
// FPU MISC 模块完整测试向量表 (RISC-V 32-bit F 扩展)
// =====================================================================
// 格式说明:
// OP_TYPE    RM   DATA1     DATA2     DATA3     EXP_RESULT EXP_FLAGS
// (字符串)   (Hex)(Hex)     (Hex)     (Hex)     (Hex)      (Hex)
// ---------------------------------------------------------------------

// ==========================================
// 1. FSGNJ (Sign Injection) 符号注入
// RM=0(fsgnj 复制), RM=1(fsgnjn 取反), RM=2(fsgnjx 异或)
// ==========================================
// fsgnj:  -1.0 (BF800000), 2.0 (40000000) -> 1.0 (3F800000)
FSGNJ         0    BF800000  40000000  00000000  3F800000   00
// fsgnjn: -1.0 (BF800000), 2.0 (40000000) -> -1.0(BF800000)
FSGNJ         1    BF800000  40000000  00000000  BF800000   00
// fsgnjx: -1.0 (BF800000), 2.0 (40000000) -> -1.0(BF800000)
FSGNJ         2    BF800000  40000000  00000000  BF800000   00
// fsgnj (NaN & Inf): qNaN (7FC00000), -Inf (FF800000) -> -qNaN (FFC00000)
FSGNJ         0    7FC00000  FF800000  00000000  FFC00000   00
// fsgnjx (NaN & NaN): qNaN (7FC00000), -qNaN (FFC00000) -> -qNaN (FFC00000)
FSGNJ         2    7FC00000  FFC00000  00000000  FFC00000   00

// ==========================================
// 2. FCMP (Compare) 浮点比较
// RM=0(fle <=), RM=1(flt <), RM=2(feq ==)
// ==========================================
// feq: 1.0 == 1.0 -> True (1)
FCMP          2    3F800000  3F800000  00000000  00000001   00
// flt: -1.0 < 1.0 -> True (1)
FCMP          1    BF800000  3F800000  00000000  00000001   00
// fle: 1.0 <= 1.0 -> True (1)
FCMP          0    3F800000  3F800000  00000000  00000001   00
// feq: +0.0 == -0.0 -> True (1)
FCMP          2    00000000  80000000  00000000  00000001   00
// flt: -0.0 < +0.0 -> False (0)
FCMP          1    80000000  00000000  00000000  00000000   00
// fle: +Inf <= +Inf -> True (1)
FCMP          0    7F800000  7F800000  00000000  00000001   00
// feq 异常检测: qNaN == 1.0 -> False(0), 无异常 (00)
FCMP          2    7FC00000  3F800000  00000000  00000000   00
// feq 异常检测: sNaN == 1.0 -> False(0), 触发 NV (10)
FCMP          2    7FA00000  3F800000  00000000  00000000   10
// flt 异常检测: qNaN < 1.0 -> False(0), 触发 NV (10) (注意: flt遇到qNaN也报NV)
FCMP          1    7FC00000  3F800000  00000000  00000000   10
// fle 异常检测: 1.0 <= sNaN -> False(0), 触发 NV (10)
FCMP          0    3F800000  7FA00000  00000000  00000000   10

// ==========================================
// 3. FMAX / FMIN 极值运算
// RM=0(fmin), RM=1(fmax)
// ==========================================
// fmin: min(1.0, -2.0) -> -2.0 (C0000000)
FMAX          0    3F800000  C0000000  00000000  C0000000   00
// fmax: max(1.0, -2.0) -> 1.0 (3F800000)
FMAX          1    3F800000  C0000000  00000000  3F800000   00
// fmin (0处理): min(+0.0, -0.0) -> -0.0 (80000000)
FMAX          0    00000000  80000000  00000000  80000000   00
// fmax (0处理): max(+0.0, -0.0) -> +0.0 (00000000)
FMAX          1    00000000  80000000  00000000  00000000   00
// fmax (qNaN容错): max(qNaN, 1.0) -> 1.0 (3F800000), 无异常
FMAX          1    7FC00000  3F800000  00000000  3F800000   00
// fmax (qNaN容错): max(1.0, qNaN) -> 1.0 (3F800000), 无异常
FMAX          1    3F800000  7FC00000  00000000  3F800000   00
// fmin (sNaN容错): min(sNaN, 1.0) -> 1.0 (3F800000), 触发 NV (10)
FMAX          0    7FA00000  3F800000  00000000  3F800000   10
// fmax (双NaN): max(qNaN, -qNaN) -> canonical qNaN (7FC00000)
FMAX          1    7FC00000  FFC00000  00000000  7FC00000   00

// ==========================================
// 4. FCLASS 浮点分类 (10个状态位 One-hot编码输出)
// RM=0 (FCLASS 不产生任何异常)
// ==========================================
// bit 0: -Inf (FF800000) -> 0x001
FCLASS        0    FF800000  00000000  00000000  00000001   00
// bit 1: -Normal (-1.0: BF800000) -> 0x002
FCLASS        0    BF800000  00000000  00000000  00000002   00
// bit 2: -Subnormal (80000001) -> 0x004
FCLASS        0    80000001  00000000  00000000  00000004   00
// bit 3: -Zero (-0.0: 80000000) -> 0x008
FCLASS        0    80000000  00000000  00000000  00000008   00
// bit 4: +Zero (+0.0: 00000000) -> 0x010
FCLASS        0    00000000  00000000  00000000  00000010   00
// bit 5: +Subnormal (00000001) -> 0x020
FCLASS        0    00000001  00000000  00000000  00000020   00
// bit 6: +Normal (1.0: 3F800000) -> 0x040
FCLASS        0    3F800000  00000000  00000000  00000040   00
// bit 7: +Inf (7F800000) -> 0x080
FCLASS        0    7F800000  00000000  00000000  00000080   00
// bit 8: sNaN (7FA00000) -> 0x100
FCLASS        0    7FA00000  00000000  00000000  00000100   00
// bit 9: qNaN (7FC00000) -> 0x200
FCLASS        0    7FC00000  00000000  00000000  00000200   00
