// =====================================================================
// FPU FMA 模块测试向量表 (RISC-V 32-bit F 扩展)
// =====================================================================
// 格式说明:
// OP_TYPE    RM   DATA1     DATA2     DATA3     EXP_RESULT EXP_FLAGS
// (字符串)   (Hex)(Hex)     (Hex)     (Hex)     (Hex)      (Hex)
// ---------------------------------------------------------------------
// IEEE-754 常数参考:
// 1.0 = 3F800000,  2.0 = 40000000, 3.0 = 40400000, 4.0 = 40800000
// -1.0= BF800000, -2.0 = C0000000
// +Inf= 7F800000, -Inf= FF800000
// qNaN= 7FC00000, sNaN= 7FA00000
// ---------------------------------------------------------------------

// ==========================================
// 1. FADD / FSUB (加减法)
// ==========================================
// 1.0 + 2.0 = 3.0
FADD          0    3F800000  40000000  00000000  40400000   00
// 1.0 - 2.0 = -1.0
FSUB          0    3F800000  40000000  00000000  BF800000   00
// 异常: +Inf + (-Inf) -> qNaN, 触发 NV 异常 (5'b10000 -> 10 hex)
FADD          0    7F800000  FF800000  00000000  7FC00000   10
// 异常: +Inf - (+Inf) -> qNaN, 触发 NV 异常
FSUB          0    7F800000  7F800000  00000000  7FC00000   10
// 异常: 遭遇 sNaN -> qNaN, 触发 NV 异常
FADD          0    7FA00000  3F800000  00000000  7FC00000   10

// ==========================================
// 2. FMUL (乘法)
// ==========================================
// 2.0 * 3.0 = 6.0 (40C00000)
FMUL          0    40000000  40400000  00000000  40C00000   00
// 异常: 0.0 * (+Inf) -> qNaN, 触发 NV 异常
FMUL          0    00000000  7F800000  00000000  7FC00000   10
// 容错: qNaN * 2.0 -> qNaN (无异常, RM=0)
FMUL          0    7FC00000  40000000  00000000  7FC00000   00

// ==========================================
// 3. FMADD: A * B + C
// ==========================================
// 2.0 * 3.0 + 4.0 = 10.0 (41200000)
FMADD         0    40000000  40400000  40800000  41200000   00
// 异常: (0.0 * Inf) + 1.0 -> qNaN, 触发 NV 异常
FMADD         0    00000000  7F800000  3F800000  7FC00000   10
// 异常: (+Inf * 1.0) + (-Inf) -> +Inf - Inf -> qNaN, 触发 NV 异常
FMADD         0    7F800000  3F800000  FF800000  7FC00000   10

// ==========================================
// 4. FMSUB: A * B - C
// ==========================================
// 2.0 * 3.0 - 4.0 = 2.0 (40000000)
FMSUB         0    40000000  40400000  40800000  40000000   00
// -2.0 * 3.0 - (-1.0) = -5.0 (C0A00000)
FMSUB         0    C0000000  40400000  BF800000  C0A00000   00

// ==========================================
// 5. FNMADD: -(A * B + C)
// ==========================================
// -(2.0 * 3.0 + 4.0) = -10.0 (C1200000)
FNMADD        0    40000000  40400000  40800000  C1200000   00

// ==========================================
// 6. FNMSUB: -(A * B - C)
// ==========================================
// -(2.0 * 3.0 - 4.0) = -2.0 (C0000000)
FNMSUB        0    40000000  40400000  40800000  C0000000   00
