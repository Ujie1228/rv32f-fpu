// =====================================================================
// FPU DIV (除法/平方根) 模块测试向量表 (RISC-V 32-bit F 扩展)
// =====================================================================
// 格式说明:
// OP_TYPE    RM   DATA1     DATA2     DATA3     EXP_RESULT EXP_FLAGS
// (字符串)   (Hex)(Hex)     (Hex)     (Hex)     (Hex)      (Hex)
// ---------------------------------------------------------------------
// 异常标志位(Flags)对应 Hex 值:
// NV (无效)   = 10
// DZ (除零)   = 08
// OF (溢出)   = 04
// UF (下溢)   = 02
// NX (不精确) = 01
// ---------------------------------------------------------------------

// ==========================================
// 1. FDIV (浮点除法)
// ==========================================
// 常规: 6.0 / 3.0 = 2.0 (40C00000 / 40400000 -> 40000000)
FDIV          0    40C00000  40400000  00000000  40000000   00
// 常规: 1.0 / 1.0 = 1.0
FDIV          0    3F800000  3F800000  00000000  3F800000   00
// 除零: 1.0 / +0.0 = +Inf, 触发 DZ (08)
FDIV          0    3F800000  00000000  00000000  7F800000   08
// 除零: -1.0 / +0.0 = -Inf, 触发 DZ (08)
FDIV          0    BF800000  00000000  00000000  FF800000   08
// 无效: +0.0 / +0.0 = qNaN, 触发 NV (10)
FDIV          0    00000000  00000000  00000000  7FC00000   10
// 无效: +Inf / +Inf = qNaN, 触发 NV (10)
FDIV          0    7F800000  7F800000  00000000  7FC00000   10
// 无效: sNaN / 1.0 = qNaN, 触发 NV (10)
FDIV          0    7FA00000  3F800000  00000000  7FC00000   10
// 容错: qNaN / 1.0 = qNaN, 无异常 (00)
FDIV          0    7FC00000  3F800000  00000000  7FC00000   00

// ==========================================
// 2. FSQRT (浮点平方根)
// FSQRT 只有 DATA1 有效，DATA2 和 DATA3 填 0
// ==========================================
// 常规: sqrt(4.0) = 2.0 (40800000 -> 40000000)
FSQRT         0    40800000  00000000  00000000  40000000   00
// 零处理: sqrt(+0.0) = +0.0
FSQRT         0    00000000  00000000  00000000  00000000   00
// 零处理: sqrt(-0.0) = -0.0
FSQRT         0    80000000  00000000  00000000  80000000   00
// 无穷大: sqrt(+Inf) = +Inf
FSQRT         0    7F800000  00000000  00000000  7F800000   00
// 无效 (负数开方): sqrt(-1.0) = qNaN, 触发 NV (10)
FSQRT         0    BF800000  00000000  00000000  7FC00000   10
// 无效: sqrt(-Inf) = qNaN, 触发 NV (10)
FSQRT         0    FF800000  00000000  00000000  7FC00000   10
// 无效: sqrt(sNaN) = qNaN, 触发 NV (10)
FSQRT         0    7FA00000  00000000  00000000  7FC00000   10
// 容错: sqrt(qNaN) = qNaN, 无异常 (00)
FSQRT         0    7FC00000  00000000  00000000  7FC00000   00
